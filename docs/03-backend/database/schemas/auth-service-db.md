    p_token_hash VARCHAR,
    p_user_id UUID
)
RETURNS boolean AS $$
DECLARE
    token_record RECORD;
BEGIN
    -- Check token exists and not expired
    SELECT expires_at, user_id
    INTO token_record
    FROM auth_tokens
    WHERE access_token = p_token_hash
    AND user_id = p_user_id;

    IF NOT FOUND THEN
        RETURN FALSE;
    END IF;

    RETURN token_record.expires_at > NOW();
END;
$$ LANGUAGE plpgsql;
